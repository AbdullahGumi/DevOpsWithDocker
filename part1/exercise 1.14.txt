#example-frontend
 
# commands used 

docker build . -t example-frontend && docker run -it -p 5000:5000 example-frontend

#Dockerfile

FROM node:14

EXPOSE 5000

WORKDIR /usr/src/app

COPY . .

ENV REACT_APP_BACKEND_URL http://localhost:8080/

RUN node -v && npm -v

RUN npm install 

RUN npm install -g serve 

RUN npm run build 

CMD ["serve", "-s", "-l", "5000", "build"]
